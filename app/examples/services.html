iceCaret

Service Info

To set the caret (cursor) position of a DOM element. This service is best used from a directive.
The 'setPosition()' code was almost entirely taken from here (thanks again stackoverflow!).

Available since

0.3.0

Usage

        
            angular
            .module('myModule')
            .directive('myDirective', function(iceCaret) {
            return {
            restrict: 'A',
            link: function(scope, element, attrs) {
            ...
            var integerTargetPosition = ...
            iceCaret.setPosition(element[0], integerTargetPosition);
            ...
            iceCaret.setPositionToEnd(element[0]);
            ...
            }
            };
            });

iceStorage

Service Info

Convenience service to use the localStorage or sessionStorage of your browser.

Available since

0.4.0

Usage local storage

        
            if (iceStorage.local.isSupported()) {
            
            iceStorage.local.set('YourStorageKey', 'some string');
            var storageValue = iceStorage.local.get('YourStorageKey');
            
            iceStorage.local.setObject('YourStorageKey', {...});
            var storageObject = iceStorage.local.getObject('YourStorageKey');
            
            iceStorage.local.remove('YourStorageKey');
            }

Usage session storage

        
            if (iceStorage.session.isSupported()) {
            
            iceStorage.session.set('YourStorageKey', 'some string');
            var storageValue = iceStorage.session.get('YourStorageKey');
            
            iceStorage.session.setObject('YourStorageKey', {...});
            var storageObject = iceStorage.session.getObject('YourStorageKey');
            
            iceStorage.session.remove('YourStorageKey');
            }
